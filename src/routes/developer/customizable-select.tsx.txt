import { createFileRoute } from "@tanstack/react-router";
import { RiMenuLine, RiMenu3Line, RiSettings4Line } from "@remixicon/react";
import { Button } from "@/components/ui/button";
import { cn } from "@/lib/utils";
import {
  Drawer,
  DrawerTrigger,
  DrawerContent,
  DrawerHeader,
  DrawerTitle,
  DrawerDescription,
  DrawerSwipeArea,
  createDrawerHandle,
} from "@/components/ui/drawer";

// Define the shape of metadata passed to the action menu workflow
interface MenuPayload {
  viewMode: "navigation" | "quick-actions" | "settings";
  title: string;
}

export const Route = createFileRoute("/developer/customizable-select")({
  component: RouteComponent,
});

const mobileMenuHandle = createDrawerHandle<MenuPayload>();

function RouteComponent() {
  const isSuperSmall = false; // or responsive hook
  const classNames = { cta: { global: "", link: "" } };

  return (
    <div>
      <p>chrome 150 and safari 27</p>
      <a href="https://www.youtube.com/watch?v=ACsHgWwP9Io">
        WWDC26: Rediscover the HTML select element | Apple
      </a>
      <select>
        <option>option 1</option>
        <option>option 2</option>
        <option>option 3</option>
      </select>

      <div className="relative min-h-screen w-full">
        {/* 2. SWIPE AREA (Invisible interaction track sitting on the right edge)
        Allows mobile users to flick right edge to snap menu out
      */}
        <DrawerSwipeArea
          handle={mobileMenuHandle}
          payload={{ viewMode: "navigation", title: "Global Explorer" }}
          className="hover:bg-primary/5 active:bg-primary/10" // Optional: visual subtle highlight guide
        />

        <header className="flex items-center justify-between p-4 border-b">
          <span>Brand Logo</span>

          <div className="flex gap-2">
            {/* Secondary trigger example: Action item modifying state via payload handles */}
            <Button
              variant="ghost"
              size="icon-sm"
              onClick={() =>
                mobileMenuHandle.openWithPayload({ viewMode: "settings", title: "Preferences" })
              }
            >
              <RiSettings4Line className="size-4" />
            </Button>

            {/* 3. MAIN TRIGGER BUTTON
            Passing the handle explicitly connects it regardless of nesting depth.
          */}
            <DrawerTrigger
              handle={mobileMenuHandle}
              payload={{ viewMode: "navigation", title: "Main Directory" }}
              render={() => (
                <Button
                  className={cn(classNames?.cta?.global, classNames?.cta?.link)}
                  variant="ghost"
                  size={isSuperSmall ? "sm" : "icon-sm"}
                />
              )}
            >
              <span className="sr-only">Menu</span>
              {isSuperSmall ? (
                <RiMenuLine className="size-4" />
              ) : (
                <RiMenu3Line className="size-4" />
              )}
              {isSuperSmall && <span>Menu</span>}
            </DrawerTrigger>
          </div>
        </header>

        {/* 4. DYNAMIC ACTION SHEET DRAWER
        Using Function-as-a-Child to safely adjust look & logic on the fly.
      */}
        <Drawer handle={mobileMenuHandle} swipeDirection="right">
          {({ payload }) => (
            <DrawerContent>
              <DrawerHeader>
                {/* Renders dynamic title based on what trigger/swipe area fired it */}
                <DrawerTitle>{payload?.title ?? "Navigation Menu"}</DrawerTitle>
                <DrawerDescription>
                  Swipe left to minimize or choose a directory step below.
                </DrawerDescription>
              </DrawerHeader>

              {/* Dynamic content rendering blocks mapped by active payload viewMode */}
              <div className="flex-1 p-4 overflow-y-auto">
                {payload?.viewMode === "navigation" && (
                  <nav className="flex flex-col gap-2">
                    <a href="#dashboard" className="p-3 rounded-md hover:bg-accent font-medium">
                      Dashboard
                    </a>
                    <a href="#analytics" className="p-3 rounded-md hover:bg-accent font-medium">
                      Analytics
                    </a>
                    <a href="#projects" className="p-3 rounded-md hover:bg-accent font-medium">
                      Projects
                    </a>
                  </nav>
                )}

                {payload?.viewMode === "settings" && (
                  <div className="flex flex-col gap-4">
                    <div className="flex justify-between items-center p-2 border-b">
                      <span>Dark Theme Mode</span>
                      <input type="checkbox" className="toggle" />
                    </div>
                    <div className="flex justify-between items-center p-2 border-b">
                      <span>Push Notifications</span>
                      <input type="checkbox" className="toggle" defaultChecked />
                    </div>
                  </div>
                )}
              </div>
            </DrawerContent>
          )}
        </Drawer>
      </div>
    </div>
  );
}
